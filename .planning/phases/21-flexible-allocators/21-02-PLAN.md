```orhon
// Allocators — collections accept optional allocator via .new(alloc)

pub func allocator_demo() i32 {
    // Mode 1: default SMP allocator (no argument)
    var defaults: List(i32) = List(i32).new()
    defaults.add(1)
    defer { defaults.free() }

    // Mode 2: inline arena allocator
    var arena: allocator.Arena = allocator.Arena.create()
    defer { arena.deinit() }
    var arena_list: List(i32) = List(i32).new(arena.allocator())
    arena_list.add(2)

    // Mode 3: external allocator variable
    var smp: allocator.SMP = allocator.SMP.create()
    defer { smp.deinit() }
    var a = smp.allocator()
    var ext_list: List(i32) = List(i32).new(a)
    ext_list.add(3)
    defer { ext_list.free() }

    return defaults.get(0) + arena_list.get(0) + ext_list.get(0)
}
```
